const OktaAPI = require('./api/okta.api');
 
const { PermissionDeniedError } = require('../../../errors/logicErrors').general;
 
const logger = require('cf-logs')
    .Logger('codefresh:github');
const monitor = require('cf-monitor');
const Promise = require('bluebird');
 
const NON_ACTIVE_USER_STATUSES = ['DEPROVISIONED'];
 
 
function adaptResult(result) {
    return result
        .map((g) => {
            const members = g.members
                .filter(m => NON_ACTIVE_USER_STATUSES.indexOf(m.status) === -1)
                .map((m) => {
                    return {
                        userName: m.profile.email.toLowerCase(),
                        email: m.profile.email.toLowerCase(),
                        roles: ['User'],
                    };
                });
            return {
                id: g.group.id,
                refId: g.group.id,
                name: g.group.profile.name,
                source: 'okta',
                members,
            };
        });
}
 
/**
 * Class for imports groups and contacts from Okta for specific account
 */
class OktaGroupImporter {
 
 
    /**
     * Validate configuration
     * Must be present keys : access_token, clientHost
     * @param config
     */
    static validate(config) {
        return config.access_token && config.clientHost && config.appId;
    }
 
    /**
     * Import groups and members per group
     * @param config - config for get access token and client host
     * @returns {Promise<*>}
     */
    static async import(config) {
        try {
            const assignments = await OktaAPI.getAssignments(config.clientHost, config.access_token, config.appId);
            const filteredAssignments = assignments.filter(a => a._links.group)
                .map(a => a._links.group.href);
            const groups = await OktaAPI.getTeams(config.clientHost, config.access_token, filteredAssignments);
            const fullGroups = await Promise.all(groups.map(async (g) => {
                const members = await OktaAPI.getMembers(g._links.users.href, config.access_token);
                return { group: g, members };
            }));
            return adaptResult(fullGroups);
        } catch (e) {
            logger.log('error', e.toString());
            monitor.noticeError(e);
            throw new PermissionDeniedError({
                message: e.message,
            });
        }
 
    }
 
}
 
module.exports = OktaGroupImporter;
