const rp = require('request-promise');
 
class AzureAPI {
 
    /**
     * Get user access token by {@param clientId}, secret {@param clientSecret}
     * and with tenant {@param tenant}
     * @param clientId - your Azure AD clientId
     * @param clientSecret - your Azure AD clientSecret
     * @param tenant - your Azure Ad tenant, Example YOURDIRECTORYNAME.onmicrosoft.com
     * @returns {Promise<void>}
     */
    static getAccessToken(clientId, clientSecret, tenant) {
        const options = {
            method: 'POST',
            uri: `https://login.microsoftonline.com/${tenant}/oauth2/v2.0/token`,
            formData: {
                grant_type: 'client_credentials',
                client_secret: clientSecret,
                client_id: clientId,
                scope: 'https://graph.microsoft.com/.default',
 
            },
            json: true,
        };
        return this.execute(options);
    }
 
    /**
     * Get logged in user groups by token {@param accessToken}
     * @param accessToken - generated token {@link getAccessToken}
     * @param limit - batch size
     * @returns [] list of groups
     */
    static async getGroups(accessToken, limit = 100) {
        const result = [];
        let groups = {};
        do {
            const uri = groups['@odata.nextLink'] ? groups['@odata.nextLink'] : `https://graph.microsoft.com/v1.0/groups?$top=${limit}`;
            const options = {
                method: 'GET',
                uri,
                auth: {
                    'bearer': accessToken,
                },
                json: true,
            };
            groups = await this.execute(options); // eslint-disable-line
            result.push(...groups.value);
        }
        while (groups['@odata.nextLink']);
        return result;
    }
 
    /**
     * Get members in group with id {@param groupId}  by token {@param accessToken}
     * @param accessToken - generated token {@link getAccessToken}
     * @param groupId - group from which we want get members
     * @returns {Promise<void>}
     */
    static getMembers(accessToken, groupId) {
        const options = {
            method: 'GET',
            uri: `https://graph.microsoft.com/v1.0/groups/${groupId}/members`,
            auth: {
                'bearer': accessToken,
            },
            json: true,
        };
        return this.execute(options);
    }
 
    /**
     * Get groups and users assigned to application {@param applicationId}
     * @param accessToken
     * @param applicationId
     */
    static getAppRoleAssignments(accessToken, applicationId) {
        const options = {
            method: 'GET',
            uri: `https://graph.microsoft.com/beta/servicePrincipals/${applicationId}/appRoleAssignments`,
            auth: {
                'bearer': accessToken,
            },
            json: true,
        };
        return this.execute(options);
    }
 
 
    static execute(options) {
        return rp(options);
    }
 
 
}
 
 
module.exports = AzureAPI;
