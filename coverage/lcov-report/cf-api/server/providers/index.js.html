const _ = require('lodash');
const { getAccountContextByName } = require('./../api/contexts/logic');

function buildContextFromIDPAndUser(user, idp) {
    const authBuilder = {
        github: () => {
            return {
                password: _.get(user, 'logins[0].credentials.accessToken'),
                permissions: _.get(user, 'logins[0].credentials.permissions'),
            };
        },
        gitlab: () => {
            return {
                username: 'oauth2', // need to do the actual clone
                password: _.get(user, 'logins[0].credentials.accessToken'),
                permissions: _.get(user, 'logins[0].credentials.permissions'),
            };
        },
        bitbucket: () => {
            return {
                username: 'x-token-auth', // need to do the actual clone
                password: _.get(user, 'logins[0].credentials.accessToken'),
                refreshToken: _.get(user, 'logins[0].credentials.refreshToken'),
                expires: _.get(user, 'logins[0].credentials.expires'),
                permissions: _.get(user, 'logins[0].credentials.permissions'),
            };
        }
    };
    const buildFn = authBuilder[idp.clientType] || _.identity;
    return {
        metadata: {
            name: 'CF-default'
        },
        spec: {
            type: `git.${idp.clientType}`,
            data: {
                auth: Object.assign({
                    idpId: 'fake-idp-id',
                    type: 'oauth2'
                }, buildFn())
            }
        },
    };
}
 
// Get the correct provider for a specific user
// TODO once we remove all personal tokens from the user it self this and all calls to it should be removed
const get = function (user) {
    return Promise.resolve()
        .then(async () => { // eslint-disable-line
            if (_.get(user, 'logins[0].credentials.accessToken')) {
                const idp = await idpLogic.findById(_.get(user, 'logins[0].idp'), true);
                const context = buildContextFromIDPAndUser(user, idp);
                const Provider = getProviderModule(idp.clientType);
                const provider = new Provider(user, context, idp);
                return provider.init(context);
            }
        });
};
 
// Get the features of a specific provider
function getFeatures(providerName) {
    const Provider = getProviderModule(providerName);
    return Provider.prototype.features;
}
 
/**
 * returns a git provider by using a specific context
 * @param user
 * @param context - a full context object
 * @returns {*}
 */
async function getProviderFromContext(user, context) {
    const provider = context.spec.type.replace('git.', '');
    const Provider = getProviderModule(provider);
    let idp;
    if (_.get(context, 'spec.data.auth.idpId')) {
        idp = await idpLogic.findById(_.get(context, 'spec.data.auth.idpId'), true);
    }
    const p = new Provider(user, context, idp);
    return p.init(context);
}
 
/**
 * returns the user provider.
 * in case of passing a context name it will go and fetch that context
 * @param user
 * @param contextName
 * @returns {Bluebird<any>}
 */
function getProvider(user, contextName, privileged = false) {
    const accountId = user.getActiveAccount()._id.toString();
    if (!contextName) {
        return Promise.resolve()
        .then(() => get(user));
    } else {
        return Promise.resolve()
        .then(() => getAccountContextByName(accountId, contextName, privileged))
        .then((context) => {
            if (!context) {
                return get(user);
            } else {
                return getProviderFromContext(user, context);
            }
        });
    }
}
