'use strict';
 
const httpInfraRequest = require('@codefresh-io/http-infra').request;
const httpInfraHelper = require('./../helpers/httpInfraHelper');
const config = require('./../../config/environment');
const rp = require('request-promise');
const accounts = require('./../accounts/accounts');
 
const clusterProvidersService = config.clusterProvidersService;
const clusterProvidersServiceUrl = `${clusterProvidersService.protocol}://${clusterProvidersService.uri}:${clusterProvidersService.port}`;
const { FailedToGetAccountClusterCredentialsError } = require('./../../../errors/logicErrors/clusters/FailedToGetAccountClusterCredentialsError');
 
function _buildUrl(accountId, selector) {
    return `${clusterProvidersServiceUrl}/api/internal/accounts/${accountId}/clusters/${selector}/credentials`;
}
 
function getClusterCredentialsForUser(user, selector) {
    const options = {
        requestOptions: {
            headers: httpInfraHelper.prepareHeadersForUser(user),
        },
        path: `/api/internal/accounts/${user.getActiveAccount()
            ._id
            .toString()}/clusters/${selector}/credentials`,
        serviceConfig: config.clusterProvidersService,
    };
 
    return httpInfraRequest(options);
}
 
function getClusterCredentialsForAccountId(accountId, selector, abacQuery) {
    const opt = {
        uri: _buildUrl(accountId, selector),
        headers: {
            'x-account-id': accountId,
        },
        json: true,
        resolveWithFullResponse: true,
        qs: abacQuery || {},
    };
    return rp(opt)
        .then((res) => { // jshint ignore:line
            return res.body;
        })
        .catch((res) => {
            throw new FailedToGetAccountClusterCredentialsError(res.error, {
                context: {
                    accountId,
                    cluster: selector,
                },
            });
        });
}
 
function getClustersForAccount(account) {
 
    const options = {
        serviceConfig: config.clusterProvidersService,
        path: '/api/clusters',
        requestOptions: {
            headers: httpInfraHelper.prepareHeadersForAccount(account),
            resolveWithFullResponse: false,
        },
    };
    return httpInfraRequest(options)
        .then(res => res.body);
 
}
 
function removeAllClustersForAccount(account, provider) {
    const accountId = account._id.toString();
    return accounts.getAccountById(accountId)
            .then((acc) => {
                if (provider === 'aks') {
                    accounts.removeAksIntegrationFromAccount(acc);
                } else if (provider === 'gcloud') {
                    accounts.removeGkeIntergrationFromAccount(acc);
                }
            })
            .then(() => {
                const options = {
                    requestOptions: {
                        method: 'DELETE',
                        headers: httpInfraHelper.prepareHeadersForAccount(account),
                    },
                    path: `/api/clusters/${accountId}/${provider}`,
                    serviceConfig: config.clusterProvidersService,
                };
                return httpInfraRequest(options);
            });
}
 
function getTags(clusterId, account) {
    const options = {
        serviceConfig: config.clusterProvidersService,
        path: `/api/clusters/${clusterId}/getTags`,
        requestOptions: {
            headers: httpInfraHelper.prepareHeadersForAccount(account),
            resolveWithFullResponse: false,
        },
    };
    return httpInfraRequest(options)
        .then(res => res.body);
}
 
function setTags(clusterId, account, tags) {
    const options = {
        serviceConfig: config.clusterProvidersService,
        path: `/api/clusters/${clusterId}/setTags`,
        requestOptions: {
            headers: httpInfraHelper.prepareHeadersForAccount(account),
            method: 'PUT',
            resolveWithFullResponse: false,
            body: tags,
        },
    };
    return httpInfraRequest(options).then(res => res.body);
}
 
function deleteCluster(clusterId, provider, user) {
    const options = {
        serviceConfig: config.clusterProvidersService,
        path: `/api/${provider}/cluster/${clusterId}`,
        requestOptions: {
            headers: httpInfraHelper.prepareHeadersForUser(user),
            method: 'DELETE',
        },
    };
    return httpInfraRequest(options).then(res => res.body);
}
 
module.exports = {
    getClusterCredentialsForUser,
    getClusterCredentialsForAccountId,
    getClustersForAccount,
    removeAllClustersForAccount,
    getTags,
    setTags,
    deleteCluster,
};
