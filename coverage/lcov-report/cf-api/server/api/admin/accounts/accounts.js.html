function transformAccountName(name) {
    return name.toLowerCase()
        .replace(/[-]+/g, '-')
        .replace(/[_]+/g, '_')
        .replace(/[.]+/g, '.')
        .replace(/[^-_.a-z0-9]+/g, '-')
        .replace(/^[_.-]+/g, '')
        .replace(/[_.-]+$/g, '');
}

function getAllAccounts(query) {
    const queryObj = queryBuilder.buildSearchQuery(query, Account, true);
    return dal.find(Account, queryObj)
        .then((allAccounts) => {
            const accountIds = _.map(allAccounts.docs, (account) => {
                return account._id.toString();
            });
            return getAllRuntimeEnvironments({ accountIds, onlyComplete: true })
                .then((runtimeEnvironments) => {
                    _.forEach(allAccounts.docs, (account) => {
                        const id = account._id.toString();
                        const accountRuntime = _.find(runtimeEnvironments, { id });
                        _.set(account, '_doc.runtimeEnvironments', _.get(accountRuntime, 'runtimeEnvironments') || []);
                    });
                    return allAccounts;
                });
        });
}

function getAccount(id) {
    return Account.findOne({ _id: id })
        .then((account) => {
            if (!account) {
                return Promise.reject(new CFError(InternalErrors.NotFoundError,
                    `could not get account id:${id}`));
            } else {
                const accountId = account._id.toString();

function updateAccount(accountId, details) {

// in case this is a private account it will also delete the owner of the account
function deleteAccount(accountId, deleteUserIfPrivateAccount, disableRestriction) {
    if (_.get(config, 'adminFeatures.deleteAccount.enabled', null) || disableRestriction) {
        let accountOwner;
        return getAccount(accountId)
            .then((account) => {
                return accounts.getUsersOfAccount(accountId, { ignorePendingUser: false })
                    .then((users) => {
                        return Promise.all(users.map((user) => {
                            if (accounts.isAccountOwner(account, user)) {
                                accountOwner = user;
                            }
                            return accounts.removeUserFromAccount(accountId, user._id, null, true);
                        }));
                    })
 
                    .then(() => {// eslint-disable-line
                        if (accountOwner && deleteUserIfPrivateAccount) {
                            return require('../user/user').deleteUser(accountOwner.userName);// eslint-disable-line
                        }
                    })
                    .then(() => {
                        return account.remove();
                    })
                    .then(() => {
                        return Promise.all(accountRelatedModels.map((model) => {
                            return model.remove({ account: accountId });
                        }));
                    });
            })
            .catch((err) => {
                return Promise.reject(new CFError(Errors.Inherit, {
                    cause: err,
                    message: `could not delete account id:${accountId.toString()}`
                }));
            });
    }    else {
        return Promise.reject(new CFError(InternalErrors.FeatureDisabled,
            'delete account feature is disabled'));
    }
}
