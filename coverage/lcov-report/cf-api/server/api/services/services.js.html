'use strict';
 
const Promise = require('bluebird'); // jshint ignore:line
const _ = require('lodash');
const CFError = require('cf-errors');
const Service = require('./services.model').Service;
const pipelinesLogic = require('../pipelines/pipeline.logic');
const repos = require('../repos/repos');
const objectHelper = require('../helpers/objectHelper');
const eventBus = require('../../eventbus');
const { DuplicationError } = require('../../../errors/logicErrors').general;
const { getProviderContextFromUser, extractContextNameFromContext } = require('./../user/users');
const { getAccountContextByName } = require('../contexts/logic');
 
const getServiceByQuery = function (query) {
    return Service.find(query);
};
 
const getServiceByName = function (user, name) {
    const queryObj = { name, account: user.getActiveAccount()._id };
    const query = Service
        .findOne(queryObj)
        .populate('account');
    const deferred = new Promise.defer();
 
    query.exec((err, service) => {
        if (err) {
            return deferred.reject(err);
        }
 
        return deferred.resolve(service);
    });
 
    return deferred.promise;
};
 
const getAccountServiceByName = function (account, name) {
    const queryObj = { name, account };
    const query = Service
        .findOne(queryObj);
    const deferred = new Promise.defer();
 
    query.exec((err, service) => {
        if (err) {
            return deferred.reject(err);
        }
 
        return deferred.resolve(service);
    });
 
    return deferred.promise;
};
 
const getServicesByContextName = function (user, name) {
    const queryObj = {
        account: user.getActiveAccount()._id,
        'context': name,
    };
 
    return getServiceByQuery(queryObj);
};
 
const createService = function (user, serviceDetails) {
    const account = user && user.getActiveAccount();
 
    if (!serviceDetails) {
        return Promise.reject(new CFError('No service details were given'));
    }
    if (!account) {
        return Promise.reject(new CFError('No account was given'));
    }
    const serviceName = _.get(serviceDetails, 'name');
    return getServiceByName(user, serviceName)
        .then((service) => {
            if (service) {
                throw new DuplicationError({
                    message: `service ${serviceName} already exists for this account`
                });
            }
 
            // make sure the user has access to this repository
            return repos.getRepo(user, serviceDetails.scm.owner.name, serviceDetails.scm.name);
        })
        .then(() => {
            serviceDetails.account = account;
            serviceDetails.scm.provider = user.provider.name;
            const contextName = extractContextNameFromContext(getProviderContextFromUser(user));
            if (contextName !== repos.DEFAULT_CONTEXT_TYPE) {
                serviceDetails.context = contextName;
            }
            return Service.create(serviceDetails).then((service) => {
                eventBus.publishEvent('service.created', {
                    aggregateId: service._id,
                    accountId: account._id,
                    props: {
                        serviceId: service._id,
                    }
                });
                return Promise.resolve(service);
            });
        });
};
 
const deleteService = function (user, name, force = false) {
    return Promise.resolve()
        .then(() => {
            const accountId = user.getActiveAccount()._id;
            return getServiceByName(user, name)
                .then((service) => {
                    //TODO remove once ABAC is introduced on the serviceObj
                    if (service.context) {
                        return getAccountContextByName(service.account._id, service.context)
                            .return(service);
                    } else {
                        return service;
                    }
                })
                .then((service) => {
                    const project = service.name;
                    return pipelinesLogic.getAllAccountPipelines(accountId, {
                        limit: 100,
                        project,
                        repoPipeline: true
                    });
                })
                .then(({ docs: pipelines }) => {
                    return Promise.all(_.map(pipelines, (pipeline) => {
                        //execute force because user can`t revoke access to triggers, and we can`t remove service in such case (SAAS-524)
                        return pipelinesLogic.deletePipelineByName(accountId, pipeline.metadata.name, force);
                    }));
                })
                .then(() => {
                    return Service.remove({ name, account: accountId });
                });
        });
};
 
const updateService = function (account, serviceName, serviceDetails) {
 
    if (!account) {
        return Promise.reject(new CFError('No account was given'));
    }
    if (!serviceName) {
        return Promise.reject(new CFError('No service was given'));
    }
 
    if (!serviceDetails) {
        return Promise.reject(new CFError('No service details were given'));
    }
 
    const accountId = account._id;
    return Service.findOne({ account: accountId, name: serviceName })
        .then((service) => {
            if (!service) {
                return Promise.reject(new CFError(`service ${serviceName} does not exist`));
            }
            return objectHelper.updateDocumentWithEnvVars(service,
                serviceDetails,
                Service,
                'environmentVariables')
                .then(() => {
                    return service.save();
                })
                .then((res) => {
                    return res;
                });
        });
};
 
module.exports = {
    create: createService,
    delete: deleteService,
    update: updateService,
    getServiceByQuery,
    getServiceByName,
    getServicesByContextName,
    getAccountServiceByName
};
